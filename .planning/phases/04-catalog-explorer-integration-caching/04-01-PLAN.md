    ```
    Module-level constants:
      _LOG_PATH = Path("logs/routing.jsonl")
      _TTL_SECONDS = 300
      _BASE_URL = "https://raw.githubusercontent.com/ianache/skills-catalog/main"

    CatalogExplorer.__init__(self, config: Config):
      self._config = config
      self._catalog_cache: tuple[float, list[dict]] | None = None

    _auth_headers(self) -> dict:
      Returns {"Authorization": f"Bearer {token}"} if config.github_token else {}

    async _fetch_catalog_yaml(self) -> list[dict]:
      GET {_BASE_URL}/catalog.yaml with _auth_headers()
      Non-200 → _log_error(url, f"HTTP {status}") → return []
      Exception → _log_error(url, str(exc)) → return []
      yaml.safe_load(resp.text) → data.get("skills", []) if isinstance(data, dict) else []
      (Note: verify actual catalog.yaml key via live fetch before assuming "skills" — check what top-level key the file uses)

    async _get_catalog(self) -> list[dict]:
      now = time.monotonic()
      if self._catalog_cache is not None and now < self._catalog_cache[0]:
          return self._catalog_cache[1]   # cache hit
      skills = await self._fetch_catalog_yaml()
      if skills:   # ONLY cache non-empty success (Pitfall 2 in RESEARCH.md)
          self._catalog_cache = (now + _TTL_SECONDS, skills)
      return skills

    _best_match(self, skills: list[dict], tags: list[str]) -> dict | None:
      OR logic: count tag overlaps per skill; return skill with highest count
      Ties resolved by catalog order (first wins)
      Return None if no skill has any overlap

    async _fetch_skill_json(self, skill_name: str) -> SkillDefinition | None:
      GET {_BASE_URL}/skills/{skill_name}/skill.json with _auth_headers()
      Non-200 → _log_error(url, ...) → return None
      Exception → _log_error(url, ...) → return None
      SkillDefinition(
          name=data["name"], description=data["description"],
          path=skill_name,   # BARE name, no 'skills/' prefix (02-01 decision in STATE.md)
          input_schema=data["input_schema"],
          allow_net_domains=data.get("allow_net_domains", []),
      )

    async find(self, tags: list[str]) -> SkillDefinition | None:
      skills = await _get_catalog()
      matched = _best_match(skills, tags)
      if matched is None: return None
      skill_name = matched.get("name") or matched.get("path", "")
      return await _fetch_skill_json(skill_name)

    async get_all_tags(self) -> list[str]:
      skills = await _get_catalog()
      all_tags = set()
      for skill in skills: all_tags.update(skill.get("tags", []))
      return sorted(all_tags)

    _log_error(self, url: str, reason: str) -> None:
      Synchronous append to logs/routing.jsonl (same pattern as _write_routing_log in agent.py)
      Record: {"type": "catalog_error", "url": url, "reason": reason, "ts": isoformat UTC}
      _LOG_PATH.parent.mkdir(parents=True, exist_ok=True)
    ```
